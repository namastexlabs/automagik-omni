CODE EXAMPLE: Conveyor IPC Pattern
================================================================================

// 1. Define schema (lib/conveyor/schemas/backend-schema.ts)
export const backendIpcSchema = {
  'send-message': {
    args: z.tuple([z.object({ text: z.string() })]),
    return: z.object({ success: z.boolean(), messageId: z.string() }),
  },
} as const

// 2. Create API class (lib/conveyor/api/backend-api.ts)
export class BackendApi extends ConveyorApi {
  sendMessage = (text: string) => this.invoke('send-message', { text })
}

// 3. Implement handler (lib/conveyor/handlers/backend-handler.ts)
handle('send-message', async ({ text }) => {
  const response = await fetch('http://localhost:8000/api/messages', {
    method: 'POST',
    body: JSON.stringify({ text }),
  })
  return await response.json()
})

// 4. Use in React (app/components/chat.tsx)
const { sendMessage } = useConveyor('backend')
const result = await sendMessage('Hello!')

THAT'S IT - Type-safe, validated, clean!
